```typescript
const loadLogs = useCallback(async () => {
  setLogsLoading(true)

    const res = await adminListLogs(logFilter)
    setLogs(res.logs || [])

const handleCleanupLogs = async () => {
  if (!cleanupConfirm) return

    const before = typeof cleanupConfirm === 'object'
      ? Date.now() - cleanupConfirm.days * 24 * 60 * 60 * 1000

    const result = await adminCleanLogs(before)
    setCleanupConfirm(null)
    toast(`已删除 ${result.deleted} 条日志`, 'success')
    await loadLogs()
  } catch (err) {
    toast(err instanceof Error ? err.message : String(err), 'error')
    setCleanupConfirm(null)
  }
}
```
